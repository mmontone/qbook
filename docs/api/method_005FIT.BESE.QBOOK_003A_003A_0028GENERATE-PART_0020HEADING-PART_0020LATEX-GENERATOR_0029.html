(defmethod generate-part ((part heading-part) (generator latex-generator))
  (write-string (ecase (depth part)
                  (1 "\\chapter{")
                  (2 "\\section{")
                  (3 "\\subsection{")
                  (4 "\\subsubsection{")
                  (5 "\\subsubsection*{"))
                *latex-stream*)
  (write-latex-escaped (text part) *latex-stream*)
  (write-string "}" *latex-stream*)
  (terpri *latex-stream*))
