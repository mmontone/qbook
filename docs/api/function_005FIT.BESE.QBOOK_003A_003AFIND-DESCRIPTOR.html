(defun find-descriptor (label name)
  (when-bind elements-of-type (gethash label *known-elements*)
    (cdr (assoc name elements-of-type
                :test (lambda (a b)
                        (if (symbolp a)
                            (eq a b)
                            (and (eq (first a) (first b))
                                 (eq (second a) (second b)))))))))
