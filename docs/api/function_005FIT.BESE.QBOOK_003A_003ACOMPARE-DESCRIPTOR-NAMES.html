(defun compare-descriptor-names (a b)
  (string< (regex-replace-all "[^A-Za-z]" (string (if (symbolp (name a))
                                                      (name a)
                                                      (second (name a)))) "")
           (regex-replace-all "[^A-Za-z]" (string (if (symbolp (name b))
                                                      (name b)
                                                      (second (name b)))) "")))
