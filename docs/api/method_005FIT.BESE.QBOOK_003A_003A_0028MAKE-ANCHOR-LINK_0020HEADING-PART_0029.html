(defmethod make-anchor-link ((h heading-part))
  (if (= 1 (depth h))
      (strcat (make-anchor-name (text h)) ".html")
      (labels ((find-level-1 (h)
                 (if (= 1 (depth h))
                     h
                     (find-level-1 (up-part h)))))
        (strcat (make-anchor-link (find-level-1 h)) "#" (make-anchor-name (text h))))))
