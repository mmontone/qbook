(defmethod generate-part-reference ((part code-part) (generator latex-generator))
  (write-source (text part) generator)
  (terpri *latex-stream*)
  (format *latex-stream* "\\hyperref[~a]{[Source Context]}"
          (descriptor-link-id (descriptor part))))
